// NATURAL VOICE ENGINE
// Google TTS (internet असल्यास) + Web Speech API fallback
// ═══════════════════════════════════════════════
var MUTED=false;
var LANG_PREF='mr'; // mr = Marathi, hi = Hindi
function toggleMute(){MUTED=!MUTED;document.getElementById('spkBtn').textContent=MUTED?'🔇':'🔊';}

// Google Translate TTS - most natural voice
var _curAudio=null;
function stopAudio(){
  if(_curAudio){try{_curAudio.pause();_curAudio.src="";}catch(e){} _curAudio=null;}
  if(window.speechSynthesis)speechSynthesis.cancel();
}
function googleTTS(text, lang){
  return new Promise(function(res,rej){
    var url='https://translate.google.com/translate_tts?ie=UTF-8&q='+encodeURIComponent(text)+'&tl='+lang+'&client=tw-ob';
    stopAudio();
    var a=new Audio(url);
    _curAudio=a;
    a.onended=function(){_curAudio=null;res();};
    a.onerror=rej;
    a.play().catch(rej);
  });
}

// Web Speech API fallback - improved natural settings
var _bestVoice=null;
function loadVoice(){
  var vs=window.speechSynthesis?speechSynthesis.getVoices():[];
  if(!vs.length)return;
  _bestVoice=
    vs.find(function(v){return v.lang==='mr-IN';}) ||
    vs.find(function(v){return v.lang==='hi-IN' && /google/i.test(v.name);}) ||
    vs.find(function(v){return v.lang==='hi-IN' && v.localService;}) ||
    vs.find(function(v){return v.lang==='hi-IN';}) ||
    vs.find(function(v){return v.lang.startsWith('hi');}) ||
    vs[0]||null;
}
if(window.speechSynthesis){
  loadVoice();
  speechSynthesis.onvoiceschanged=loadVoice;
  setTimeout(loadVoice,800);
}
function webSpeak(text, isLong){
  if(!window.speechSynthesis)return;
  speechSynthesis.cancel();
  setTimeout(function(){
    var u=new SpeechSynthesisUtterance(text);
    if(_bestVoice)u.voice=_bestVoice;
    u.lang=_bestVoice?_bestVoice.lang:'hi-IN';
    u.rate=isLong?0.72:0.82; // slower for paragraphs/kavita
    u.pitch=isLong?1.0:1.05;
    u.volume=1;
    speechSynthesis.speak(u);
  },60);
}
var _speakToken=0;
function speak(text){
  if(MUTED)return;
  if(!text)return;
  _speakToken++;
  var myToken=_speakToken;
  stopAudio();
  if(window.speechSynthesis){speechSynthesis.cancel();}
  // For long text (kavita/parichhed): try Google TTS with shorter chunk first
  // Google TTS has 200 char limit - use first meaningful sentence
  var shortText = text.length > 180 ? text.substring(0,180) : text;
  googleTTS(shortText, LANG_PREF).catch(function(){
    if(myToken!==_speakToken)return;
    if(window.speechSynthesis){speechSynthesis.cancel();}
    // Web Speech: use slower rate for long text, normal for short
    webSpeak(text, text.length > 100);
  });
}

function h(tag,props,children){
  var el=document.createElement(tag);
  if(props)Object.keys(props).forEach(function(k){
    if(k==="style"&&typeof props[k]==="object"){Object.assign(el.style,props[k]);}
    else if(k==="className"){el.className=props[k];}
    else if(k==="onclick"){el.onclick=props[k];}
    else if(k==="innerHTML"){el.innerHTML=props[k];}
    else{el.setAttribute(k,props[k]);}
  });
  if(children){
    if(!Array.isArray(children))children=[children];
    function addChild(ch){
      if(ch===null||ch===undefined)return;
      if(Array.isArray(ch)){ch.forEach(addChild);return;}
      if(typeof ch==="string"||typeof ch==="number"){el.appendChild(document.createTextNode(ch));}
      else if(ch&&ch.nodeType){el.appendChild(ch);}
    }
    children.forEach(addChild);
  }
  return el;
}
function speakSlow(txt,id){
  // For barakhaadi: speak slowly and clearly
  if(!MUTED && txt){
    _speakToken++;
    var myToken=_speakToken;
    stopAudio();
    if(window.speechSynthesis){speechSynthesis.cancel();}
    googleTTS(txt,"mr").catch(function(){
      if(myToken!==_speakToken)return;
      if(window.speechSynthesis){speechSynthesis.cancel();}
      setTimeout(function(){
        var u=new SpeechSynthesisUtterance(txt);
        if(_bestVoice)u.voice=_bestVoice;
        u.lang=_bestVoice?_bestVoice.lang:"hi-IN";
        u.rate=0.6; u.pitch=1.0; u.volume=1; // very slow and clear
        speechSynthesis.speak(u);
      },60);
    });
  }
  var prev=document.querySelector(".p");
  if(prev)prev.classList.remove("p");
  var el=document.getElementById("c_"+id);
  if(el){el.classList.add("p");}
  S.clicked=id; render();
}

function renderQuiz(){
  if(S.quizDone){
    var pct=Math.round(S.quizScore/QUIZ.filter(function(q){return q.lv===S.quizLevel;}).length*100);
    return h("div",{style:{padding:"20px",textAlign:"center"}},
      [h("div",{style:{fontSize:"64px",margin:"20px 0"}},pct>=80?"🏆":pct>=50?"🌟":"💪"),
       h("div",{style:{fontSize:"28px",fontWeight:"900",color:"#BF360C",marginBottom:"8px"}},"निकाल"),
       h("div",{style:{fontSize:"20px",color:"#333",marginBottom:"20px"}},S.quizScore+" / "+QUIZ.filter(function(q){return q.lv===S.quizLevel;}).length+" गुण ("+pct+"%)"),
       h("div",{style:{fontSize:"16px",color:pct>=80?"#4CAF50":pct>=50?"#FF9800":"#F44336",fontWeight:"800",marginBottom:"24px"}},pct>=80?"अत्युत्तम! शाब्बास!! 🎉":pct>=50?"चांगले! अजून सराव करा 👍":"पुन्हा प्रयत्न करा 💪"),
       h("button",{style:{background:"linear-gradient(135deg,#BF360C,#FF6B1A)",color:"#fff",border:"none",borderRadius:"14px",padding:"14px 32px",fontSize:"18px",fontWeight:"800",cursor:"pointer"},
         onclick:function(){S.quizIdx=0;S.quizScore=0;S.quizDone=false;render();}},
         "🔄 पुन्हा खेळा")]);
  }
  var qs=QUIZ.filter(function(q){return q.lv===S.quizLevel;});
  if(S.quizIdx>=qs.length){S.quizDone=true;render();return h("div",{});}
  var q=qs[S.quizIdx];
  var pct=(S.quizIdx/qs.length)*100;
  return h("div",{style:{padding:"14px"}},
    [h("div",{style:{fontSize:"18px",fontWeight:"900",marginBottom:"10px",color:"#BF360C"}},"🎮 मराठी सराव"),
     h("div",{style:{display:"flex",gap:"6px",marginBottom:"12px"}},
       [1,2,3].map(function(lv){
         return h("button",{className:"stab",style:{background:S.quizLevel===lv?"#BF360C":"#fff",color:S.quizLevel===lv?"#fff":"#BF360C",borderColor:"#BF360C"},
           onclick:function(){S.quizLevel=lv;S.quizIdx=0;S.quizScore=0;S.quizDone=false;render();}},
           "स्तर "+lv);
       })),
     h("div",{style:{background:"#FFF3E0",borderRadius:"10px",padding:"8px 14px",marginBottom:"16px"}},
       [h("div",{style:{display:"flex",justifyContent:"space-between",fontSize:"13px",color:"#BF360C",fontWeight:"700",marginBottom:"5px"}},
         [h("span",{},"प्रश्न "+(S.quizIdx+1)+" / "+qs.length),h("span",{},"गुण: "+S.quizScore)]),
        h("div",{className:"pbar"},h("div",{className:"pfill",style:{width:pct+"%",background:"linear-gradient(90deg,#FF6B1A,#BF360C)"}}))]),
     h("div",{style:{background:"linear-gradient(135deg,#FF6B1A,#BF360C)",borderRadius:"18px",padding:"24px",marginBottom:"16px",textAlign:"center",color:"#fff",minHeight:"100px",display:"flex",alignItems:"center",justifyContent:"center"}},
       h("div",{style:{fontSize:"22px",fontWeight:"900",lineHeight:"1.5"}},q.q)),
     h("div",{style:{display:"grid",gridTemplateColumns:"1fr 1fr",gap:"10px"}},
       q.opts.map(function(opt,idx){
         return h("button",{style:{background:"#fff",border:"2px solid #ddd",borderRadius:"14px",padding:"14px",fontSize:"16px",fontWeight:"700",cursor:"pointer",transition:"all 0.15s",fontFamily:"inherit"},
           onclick:function(){
             if(idx===q.ans)S.quizScore++;
             S.quizIdx++;
             if(idx===q.ans){this.style.background="#4CAF50";this.style.color="#fff";this.style.borderColor="#4CAF50";}
             else{this.style.background="#f44336";this.style.color="#fff";this.style.borderColor="#f44336";}
             setTimeout(function(){render();},400);
           }},opt);
       }))]);
}
